        window.onload = () => {
            getDominantImageColor = () => {
                // Obtén la imagen
                let sourceImage = document.querySelector("img");
                // Obtén el elemento de fondo
                let donut = document.querySelector(".donut");
                // Inicializa colorThief
                let colorThief = new ColorThief();
                // Obtén la paleta de colores
                let color = colorThief.getColor(sourceImage);
                console.log(color);
                // Establece el color de fondo
                donut.style.backgroundColor = "rgb(" + color[0] +"," + color[1] +","+ color[2] +")";
            }
            getDominantImageColor();
        }
